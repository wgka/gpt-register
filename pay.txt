// @name         ChatGPT 快捷支付
// @namespace    Linuxdo-daydayup
// @version      1.0
// @description  复制 Access、生成 Plus/Team 支付短链
// @author       Linuxdo-daydayup
// @match        https://chatgpt.com/*
// @match        https://*.chatgpt.com/*
// @icon         https://chatgpt.com/favicon.ico
// @grant        GM_setValue
// @grant        GM_getValue
// @grant        GM_setClipboard

(function () {
'use strict';

const STORAGE_KEY = 'tm_neon_panel_position';
const COLLAPSED_KEY = 'tm_neon_panel_collapsed';
const PANEL_ACTIONS = [
    { id: 'tm-act-token', icon: 'A', tag: 'AUTH', title: '复制 Access', accentClass: 'tm-accent-auth' },
    { id: 'tm-act-plus', icon: 'P', tag: 'PLUS', title: 'Plus 试用', accentClass: 'tm-accent-plus' },
    { id: 'tm-act-team', icon: 'T', tag: 'TEAM', title: 'Team 支付', accentClass: 'tm-accent-team' }
];

function $id(id) {
    return document.getElementById(id);
}

function copyToClipboard(text) {
    GM_setClipboard(text, 'text');
}

function clamp(value, min, max) {
    return Math.min(Math.max(value, min), max);
}

function setActionLoading(button, loading) {
    if (!button) return;
    button.disabled = loading;
    button.classList.toggle('is-loading', loading);
}

function setActionLabel(button, text) {
    const title = button?.querySelector('.tm-action-title');
    if (title) title.textContent = text;
}

function resetActionLabel(button, delayMs = 1800) {
    if (!button) return;
    setTimeout(() => {
        setActionLoading(button, false);
        setActionLabel(button, button.dataset.defaultTitle || '操作');
    }, delayMs);
}

async function getAccessToken() {
    const response = await fetch('/api/auth/session');
    if (response.status === 403) throw new Error('403 Forbidden');
    const data = await response.json();
    if (!data.accessToken) throw new Error('未登录');
    return data.accessToken;
}

function buildCheckoutUrl(checkoutSessionId) {
    return 'https://chatgpt.com/checkout/openai_llc/' + checkoutSessionId;
}

function resolveCheckoutUrl(data) {
    if (data.checkout_session_id) {
        return buildCheckoutUrl(data.checkout_session_id);
    }

    const url = typeof data.url === 'string' ? data.url : '';
    if (!url) return '';

    const shortLinkMatch = url.match(/\/checkout\/openai_llc\/([A-Za-z0-9_-]+)/);
    if (shortLinkMatch) {
        return buildCheckoutUrl(shortLinkMatch[1]);
    }

    return url;
}

async function requestCheckout(token, payload) {
    const response = await fetch('https://chatgpt.com/backend-api/payments/checkout', {
        method: 'POST',
        headers: {
            Authorization: 'Bearer ' + token,
            'Content-Type': 'application/json'
        },
        body: JSON.stringify(payload)
    });

    const data = await response.json().catch(() => ({}));
    if (!response.ok) {
        throw new Error(data.detail || data.message || ('HTTP ' + response.status));
    }

    return data;
}

function openGeneratedLink(popup, url) {
    if (popup && !popup.closed) {
        popup.location.replace(url);
        return;
    }

    window.open(url, '_blank', 'noopener');
}

async function handleCopyToken() {
    const button = $id('tm-act-token');

    try {
        setActionLoading(button, true);
        setActionLabel(button, '获取中...');
        copyToClipboard(await getAccessToken());
        setActionLabel(button, '已复制');
    } catch (error) {
        setActionLabel(button, '失败');
    }

    resetActionLabel(button);
}

async function handlePlusPayment() {
    const button = $id('tm-act-plus');
    const popup = openPendingWindow('Plus 试用');

    try {
        setActionLoading(button, true);
        setActionLabel(button, '生成中...');
        const token = await getAccessToken();
        const payload = {
            plan_name: 'chatgptplusplan',
            billing_modality: 'paid_auto_renewing',
            promo_campaign: {
                promo_campaign_id: 'plus-1-month-free',
                is_coupon_from_query_param: true
            },
            checkout_ui_mode: 'custom'
        };
        const data = await requestCheckout(token, payload);
        const url = resolveCheckoutUrl(data);
        if (!url) throw new Error('未返回 Plus 短链');

        openGeneratedLink(popup, url);
        setActionLabel(button, '已打开');
    } catch (error) {
        if (popup && !popup.closed) popup.close();
        alert('错误: ' + (error.message || error));
        setActionLabel(button, '失败');
    }

    resetActionLabel(button);
}

async function handleTeamPayment() {
    const button = $id('tm-act-team');
    const popup = openPendingWindow('Team 支付');

    try {
        setActionLoading(button, true);
        setActionLabel(button, '生成中...');
        const token = await getAccessToken();
        const payload = {
            plan_name: 'chatgptteamplan',
            team_plan_data: {
                workspace_name: 'MyTeam',
                price_interval: 'month',
                seat_quantity: 5
            },
            promo_campaign: {
                promo_campaign_id: 'team-1-month-free',
                is_coupon_from_query_param: true
            },
            checkout_ui_mode: 'custom'
        };
        const data = await requestCheckout(token, payload);
        const url = resolveCheckoutUrl(data);
        if (!url) throw new Error('未返回 Team 短链');

        openGeneratedLink(popup, url);
        setActionLabel(button, '已打开');
    } catch (error) {
        if (popup && !popup.closed) popup.close();
        alert('错误: ' + (error.message || error));
        setActionLabel(button, '失败');
    }

    resetActionLabel(button);
}

function handlePanelAction(button) {
    switch (button.id) {
        case 'tm-act-token':
            void handleCopyToken();
            break;
        case 'tm-act-plus':
            void handlePlusPayment();
            break;
        case 'tm-act-team':
            void handleTeamPayment();
            break;
    }
}

function renderActionButton(action) {
    return `
        <button class="tm-action-item" id="${action.id}" data-default-title="${action.title}">
            <div class="tm-icon-box ${action.accentClass}">${action.icon}</div>
            <div class="tm-action-title">${action.title}</div>
            <div class="tm-action-pill">${action.tag}</div>
        </button>`;
}

function setPanelPosition(root, panel, left, top, persist = false) {
    const margin = 8;
    const maxLeft = Math.max(margin, window.innerWidth - panel.offsetWidth - margin);
    const maxTop = Math.max(margin, window.innerHeight - panel.offsetHeight - margin);
    const nextLeft = clamp(left, margin, maxLeft);
    const nextTop = clamp(top, margin, maxTop);

    root.style.left = `${nextLeft}px`;
    root.style.top = `${nextTop}px`;
    root.style.right = 'auto';
    root.style.bottom = 'auto';

    if (persist) {
        GM_setValue(STORAGE_KEY, { left: nextLeft, top: nextTop });
    }
}

function restorePanelPosition(root, panel) {
    const position = GM_getValue(STORAGE_KEY, null);
    if (!position || typeof position.left !== 'number' || typeof position.top !== 'number') return;
    setPanelPosition(root, panel, position.left, position.top);
}

function enablePanelDrag(root, panel) {
    const header = panel.querySelector('.tm-panel-header');
    if (!header) return;

    let dragState = null;

    const stopDrag = () => {
        if (!dragState) return;
        header.classList.remove('is-dragging');
        document.body.style.userSelect = '';
        setPanelPosition(root, panel, parseFloat(root.style.left), parseFloat(root.style.top), true);
        dragState = null;
        window.removeEventListener('pointermove', handlePointerMove);
        window.removeEventListener('pointerup', stopDrag);
        window.removeEventListener('pointercancel', stopDrag);
    };

    const handlePointerMove = (event) => {
        if (!dragState) return;
        setPanelPosition(root, panel, event.clientX - dragState.offsetX, event.clientY - dragState.offsetY);
    };

    header.addEventListener('pointerdown', (event) => {
        if (event.button !== 0) return;
        const rect = panel.getBoundingClientRect();
        dragState = {
            offsetX: event.clientX - rect.left,
            offsetY: event.clientY - rect.top
        };
        header.classList.add('is-dragging');
        document.body.style.userSelect = 'none';
        window.addEventListener('pointermove', handlePointerMove);
        window.addEventListener('pointerup', stopDrag);
        window.addEventListener('pointercancel', stopDrag);
    });

    window.addEventListener('resize', () => {
        if (!root.style.left || !root.style.top) return;
        setPanelPosition(root, panel, parseFloat(root.style.left), parseFloat(root.style.top), true);
    });
}

function togglePanelCollapse(root, panel) {
    const isCollapsed = panel.classList.toggle('is-collapsed');
    GM_setValue(COLLAPSED_KEY, isCollapsed);
    const btn = panel.querySelector('.tm-collapse-btn');
    if (btn) btn.textContent = isCollapsed ? '+' : '-';
}

function createMainPanel() {
    if ($id('tm-app-root')) return;

    const root = document.createElement('div');
    root.id = 'tm-app-root';

    const panel = document.createElement('div');
    panel.className = 'tm-app-panel';
    panel.innerHTML = `
        <div class="tm-panel-header">
            <div class="tm-panel-brand tm-collapsed-pill">
                <div class="tm-panel-mark">P</div>
                <div class="tm-panel-title">Quick Pay</div>
            </div>
            <button class="tm-collapse-btn" title="折叠/展开">-</button>
        </div>
        <div class="tm-panel-body">
            ${PANEL_ACTIONS.map(renderActionButton).join('')}
        </div>`;

    root.appendChild(panel);
    document.body.appendChild(root);
    restorePanelPosition(root, panel);

    if (GM_getValue(COLLAPSED_KEY, false)) {
        panel.classList.add('is-collapsed');
        const btn = panel.querySelector('.tm-collapse-btn');
        if (btn) btn.textContent = '+';
    }

    enablePanelDrag(root, panel);

    panel.querySelector('.tm-collapse-btn')?.addEventListener('click', (event) => {
        event.stopPropagation();
        togglePanelCollapse(root, panel);
    });

    panel.addEventListener('click', (event) => {
        event.stopPropagation();
        const button = event.target.closest('.tm-action-item');
        if (!button) return;
        handlePanelAction(button);
    });
}
